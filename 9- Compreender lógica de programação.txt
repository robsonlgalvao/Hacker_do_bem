#Contador

#acumulador
soma=0
#repetição
while(cont<=5):
  idade=int(input(f"Digite A idade o usuário nº{cont}: "))
  soma=soma+idade #aqui acumulando
  cont=cont+1	#aqui contador
media=soma/5	#ou media=soma/(contador-1)  aqui calculo média
print(f"A média das idades é: {media}")
